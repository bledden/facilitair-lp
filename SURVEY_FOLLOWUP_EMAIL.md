## Implementation Script

Create a script to send this email to all confirmed subscribers who haven't completed the survey:

```javascript
// send-survey-followup.js
require('dotenv').config();
const Database = require('better-sqlite3');
const { Resend } = require('resend');

const resend = new Resend(process.env.RESEND_API_KEY);
const db = new Database('facilitair-emails.db');

const emailHTML = /* paste email HTML template above with {{SURVEY_URL}} placeholder */;

async function sendFollowUpEmails() {
    const subscribers = db.prepare(`
        SELECT email, unsubscribe_token
        FROM subscribers
        WHERE confirmed = 1 AND (survey_completed IS NULL OR survey_completed = 0)
    `).all();

    console.log(`Sending survey follow-up to ${subscribers.length} subscribers...`);

    for (const subscriber of subscribers) {
        const surveyUrl = `${process.env.BASE_URL}/survey.html?token=${subscriber.unsubscribe_token}`;
        const html = emailHTML.replace(/{{SURVEY_URL}}/g, surveyUrl);

        try {
            const { data, error } = await resend.emails.send({
                from: process.env.FROM_EMAIL || 'FACILITAIR <team@facilitair.ai>',
                to: subscriber.email,
                subject: 'Quick Request: Help Shape FACILITAIR\'s Beta (2 Minutes)',
                html: html
            });

            if (error) {
                console.error(`✗ Failed to send to ${subscriber.email}:`, error);
            } else {
                console.log(`✓ Sent to ${subscriber.email} (ID: ${data.id})`);
            }

            // Rate limit: wait 100ms between sends
            await new Promise(resolve => setTimeout(resolve, 100));
        } catch (error) {
            console.error(`✗ Error sending to ${subscriber.email}:`, error);
        }
    }

    console.log('Done!');
}

sendFollowUpEmails().catch(console.error);
```

Run with:
```bash
BASE_URL="https://facilitair.ai" node send-survey-followup.js
```
